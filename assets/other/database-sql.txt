CREATE TABLE usertable
(
userID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(userID),
userFName VARCHAR(48) NOT NULL,
userSName VARCHAR(48) NOT NULL,
userPassword VARCHAR(48) NOT NULL,
userLanguage VARCHAR(48) NOT NULL,
userSpecialisation VARCHAR(512) NOT NULL,
userIsAdmin INT
)

CREATE TABLE lessontable
(
lessonID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(lessonID),
lessonCategoryID INT NOT NULL,
lessonTitle VARCHAR(128) NOT NULL
)

CREATE TABLE questiontable
(
questionID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(questionID),
questionLessonID INT NOT NULL,
questionType INT NOT NULL,
questionAnswers TEXT NOT NULL,
questionAnswerIndex INT NOT NULL,
questionHelperText TEXT,
questionTimer INT
)

CREATE TABLE categorytable
(
categoryID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(categoryID),
categoryTitle VARCHAR(128) NOT NULL	
)

CREATE TABLE progressiontable
(
progressionID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(progressionID),
progressionUserID INT NOT NULL,
progressionCategoryID INT NOT NULL,
progressionLessonID INT NOT NULL,
progressionScore INT NOT NULL
)

CREATE TABLE lessonprerequisitetable
(
prereqID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(prereqID),
prereqCategoryID INT NOT NULL,
prereqUnlocksID INT NOT NULL,
prereqScore INT NOT NULL
)

CREATE TABLE rewardstable
(
rewardID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(rewardID),
rewardName VARCHAR(128) NOT NULL,
rewardCriteria VARCHAR(128) NOT NULL
)

CREATE TABLE userrewardtable
(
userrewardID INT NOT NULL AUTO_INCREMENT, PRIMARY KEY(userrewardID),
rewardedUserID INT NOT NULL,
rewardID INT NOT NULL
)
